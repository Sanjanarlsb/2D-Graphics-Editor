#include <stdio.h>

#define WIDTH 40
#define HEIGHT 20

// Picture canvas
char picture[HEIGHT][WIDTH];

// Initialize picture with spaces
void initializePicture()
{
    int i, j;

    for(i = 0; i < HEIGHT; i++)
    {
        for(j = 0; j < WIDTH; j++)
        {
            picture[i][j] = ' ';
        }
    }
}

// Display picture
void displayPicture()
{
    int i, j;

    for(i = 0; i < HEIGHT; i++)
    {
        for(j = 0; j < WIDTH; j++)
        {
            printf("%c", picture[i][j]);
        }

        printf("\n");
    }
}

int main()
{
    initializePicture();

    printf("2D Graphics Editor Started\n");

    displayPicture();

    return 0;
}
